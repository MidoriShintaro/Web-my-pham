        let count = 1;

        const decrease = document.querySelector('.decrease')
        const quantity = document.getElementById('number');

        decrease.addEventListener('click', () => {
            if (count > 0) {
                quantity.setAttribute('value', `${count -= 1}`)
            }
            temp = count;
            return temp;

        increase.addEventListener('click', () => {
            quantity.setAttribute('value', `${count += 1}`)

                    </div>
                </div>                                                
            </li>

            $('.list-cart').append(list)

            listcart.classList.add('show')
            addProduct(findId)
        })
        const removeProduct = () => {
            $('.list-product-item').remove()
            empty.classList.add('show')
            listcart.classList.remove('show')
        }
        $(document).ready(() => {
            $('.productdetail-title').html(findId.name)

            $('.productCode').html(findId.productCode)
            $('#img-product').attr('src', findId.img)
        })
